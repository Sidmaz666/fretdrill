   796→function dedupNotes(notes: FretboardNote[]): FretboardNote[] {
   797→  const seen = new Set<string>();
   798→  return notes.filter(n => {
   799→    const k = `${n.string}-${n.fret}`;
   800→    if (seen.has(k)) return false;
   801→    seen.add(k);
   802→    return true;
   803→  });
   804→}

   806→export function sortNotesAscending(notes: FretboardNote[]): FretboardNote[] {
   807→  return [...notes].sort((a, b) => {
   808→    const pitchA = (5 - a.string) * 5 + a.fret;
   809→    const pitchB = (5 - b.string) * 5 + b.fret;
   810→    if (pitchA !== pitchB) return pitchA - pitchB;
   811→    return a.fret - b.fret;
   812→  });
   813→}
   814→
   815→function sortNotesDescending(notes: FretboardNote[]): FretboardNote[] {
   816→  return [...notes].sort((a, b) => {
   817→    const pitchA = (5 - a.string) * 5 + a.fret;
   818→    const pitchB = (5 - b.string) * 5 + b.fret;
   819→    if (pitchA !== pitchB) return pitchB - pitchA;
   820→    return b.fret - a.fret;
   821→  });
   822→}

   828→function generateTabs(notes: ExerciseNote[], fretCount: number = 20): string[] {
   829→  const stringNames = ['e', 'B', 'G', 'D', 'A', 'E'];
   830→  const tabs: string[] = stringNames.map(s => s + '|');
   831→
   832→  const maxDisplayNotes = Math.min(notes.length, 20);
   833→  const displayNotes = notes.slice(0, maxDisplayNotes);
   834→
   835→  for (const note of displayNotes) {
   836→    const stringRow = 5 - note.string;
   837→    for (let row = 0; row < 6; row++) {
   838→      if (row === stringRow) {
   839→        const fretStr = note.fret.toString();
   840→        tabs[row] += '-' + fretStr + '-';
   841→      } else {
   842→        tabs[row] += '---';
   843→      }
   844→    }
   845→  }
   846→
   847→  const maxLen = Math.max(...tabs.map(t => t.length));
   848→  for (let i = 0; i < tabs.length; i++) {
   849→    while (tabs[i].length < maxLen) tabs[i] += '-';
   850→  }
   851→
   852→  return tabs;
   853→}

   855→function makeExercise(type: ExerciseType, notes: FretboardNote[], fallbackNotes: FretboardNote[]): Exercise {
   856→  const usedNotes = notes.length >= 3 ? notes : fallbackNotes;
   857→  const exerciseNotes = addSequenceNumbers(usedNotes.map(fretboardToExerciseNote));
   858→  return {
   859→    name: EXERCISE_TYPES[type].name,
   860→    description: EXERCISE_TYPES[type].description,
   861→    type,
   862→    notes: exerciseNotes,
   863→    tabs: generateTabs(exerciseNotes),
   864→  };
   865→}

   991→  return makeExercise('pedal-tone', pedalNotes, sorted);

  1068→function generateArpeggios(key: NoteName, scaleId: string, startFret: number, endFret: number): Exercise {
  1069→  const notes = getScaleOnFretboard(key, scaleId, startFret, endFret);
  1070→  const sorted = sortNotesAscending(dedupNotes(notes));
  1071→  const arpNotes: FretboardNote[] = [];
  1072→  const seen = new Set<string>();
  1073→
  1074→  for (let i = 0; i < sorted.length; i++) {
  1075→    const root = sorted[i];
  1076→    const key = `${root.string}-${root.fret}`;
  1077→    if (seen.has(key)) continue;
  1078→
  1079→    const thirdCandidates = sorted.filter(n =>
  1080→      !seen.has(`${n.string}-${n.fret}`) &&
  1081→      n !== root &&
  1082→      (n.interval === (root.interval + 3) % 12 || n.interval === (root.interval + 4) % 12) &&
  1083→      Math.abs(n.string - root.string) <= 2
  1084→    );
  1085→    const fifthCandidates = sorted.filter(n =>
  1086→      !seen.has(`${n.string}-${n.fret}`) &&
  1087→      n !== root &&
  1088→      (n.interval === (root.interval + 7) % 12) &&
  1089→      Math.abs(n.string - root.string) <= 3
  1090→    );
  1091→
  1092→    if (root.isRoot || i % 2 === 0) {
  1093→      seen.add(key);
  1094→      arpNotes.push(root);
  1095→      if (thirdCandidates.length > 0) {
  1096→        const third = thirdCandidates[0];
  1097→        seen.add(`${third.string}-${third.fret}`);
  1098→        arpNotes.push(third);
  1099→      }
  1100→      if (fifthCandidates.length > 0) {
  1101→        const fifth = fifthCandidates[0];
  1102→        seen.add(`${fifth.string}-${fifth.fret}`);
  1103→        arpNotes.push(fifth);
  1104→      }
  1105→    }
  1106→  }
  1107→
  1108→  return makeExercise('arpeggios', arpNotes, sorted);
  1109→}

  1163→function generateShellVoicings(key: NoteName, scaleId: string, startFret: number, endFret: number): Exercise {

  1165→  const sorted = sortNotesAscending(dedupNotes(notes));
  1166→  const shellNotes: FretboardNote[] = [];
  1167→  const seen = new Set<string>();
  1168→
  1169→  // Shell voicings: Root + 3rd + 7th (or b7th)
  1170→  for (let i = 0; i < sorted.length; i++) {
  1171→    const root = sorted[i];
  1172→    if (!root.isRoot && i % 3 !== 0) continue;
  1173→    const rootKey = `${root.string}-${root.fret}`;
  1174→    if (seen.has(rootKey)) continue;
  1175→
  1176→    // Find 3rd and 7th on nearby strings
  1177→    const third = sorted.find(n =>
  1178→      !seen.has(`${n.string}-${n.fret}`) &&
  1179→      n !== root &&
  1180→      (n.interval === 3 || n.interval === 4) &&
  1181→      Math.abs(n.string - root.string) <= 3 &&
  1182→      n.string > root.string
  1183→    );
  1184→    const seventh = sorted.find(n =>
  1185→      !seen.has(`${n.string}-${n.fret}`) &&
  1186→      n !== root &&
  1187→      n !== third &&
  1188→      (n.interval === 10 || n.interval === 11) &&
  1189→      Math.abs(n.string - root.string) <= 4 &&
  1190→      n.string > root.string
  1191→    );
  1192→
  1193→    if (third || seventh) {
  1194→      seen.add(rootKey);
  1195→      shellNotes.push(root);
  1196→      if (third) {
  1197→        seen.add(`${third.string}-${third.fret}`);
  1198→        shellNotes.push(third);
  1199→      }
  1200→      if (seventh) {
  1201→        seen.add(`${seventh.string}-${seventh.fret}`);
  1202→        shellNotes.push(seventh);
  1203→      }
  1204→    }
  1205→  }

  1210→function generateDyads(key: NoteName, scaleId: string, startFret: number, endFret: number): Exercise {
  1211→  const notes = getScaleOnFretboard(key, scaleId, startFret, endFret);
  1212→  const sorted = sortNotesAscending(dedupNotes(notes));
  1213→  const dyadNotes: FretboardNote[] = [];
  1214→  const seen = new Set<string>();
  1215→
  1216→  // Two-note combinations on adjacent strings
  1217→  for (let s = 0; s < 5; s++) {
  1218→    const lowerNotes = sorted.filter(n => n.string === s);
  1219→    const higherNotes = sorted.filter(n => n.string === s + 1);
  1220→    for (const ln of lowerNotes) {
  1221→      // Find closest note on next string
  1222→      const closest = higherNotes.reduce((best: FretboardNote | null, hn) => {
  1223→        if (seen.has(`${hn.string}-${hn.fret}`)) return best;
  1224→        const dist = Math.abs(hn.fret - ln.fret);
  1225→        if (!best || dist < Math.abs(best.fret - ln.fret)) return hn;
  1226→        return best;
  1227→      }, null);
  1228→      if (closest) {
  1229→        const lk = `${ln.string}-${ln.fret}`;
  1230→        const ck = `${closest.string}-${closest.fret}`;
  1231→        if (!seen.has(lk)) { seen.add(lk); dyadNotes.push(ln); }
  1232→        if (!seen.has(ck)) { seen.add(ck); dyadNotes.push(closest); }
  1233→      }
  1234→    }

  1240→function generateChordScale(key: NoteName, scaleId: string, startFret: number, endFret: number): Exercise {
