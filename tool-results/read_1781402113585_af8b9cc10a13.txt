    94‚Üí    94‚Üí    94‚Üífunction playGuitarNote(stringIdx: number, fret: number, duration: number = 0.5) {
    95‚Üí    95‚Üí    95‚Üí  try {
    96‚Üí    96‚Üí    96‚Üí    const ctx = getAudioCtx();
    97‚Üí    97‚Üí    97‚Üí    if (ctx.state === 'suspended') ctx.resume();
    98‚Üí    98‚Üí    98‚Üí    const master = getMaster(ctx);
    99‚Üí    99‚Üí    99‚Üí    const baseFreq = STRING_FREQUENCIES[stringIdx];
   100‚Üí   100‚Üí   100‚Üí    const freq = baseFreq * Math.pow(2, fret / 12);
   101‚Üí   101‚Üí   101‚Üí    const now = ctx.currentTime;
   102‚Üí   102‚Üí   102‚Üí
   103‚Üí   103‚Üí   103‚Üí    // ‚îÄ‚îÄ Fundamental (pulse wave ‚Äî close to plucked string) ‚îÄ‚îÄ
   104‚Üí   104‚Üí   104‚Üí    const osc1 = ctx.createOscillator();
   105‚Üí   105‚Üí   105‚Üí    const gain1 = ctx.createGain();
   106‚Üí   106‚Üí   106‚Üí    osc1.type = 'triangle';
   107‚Üí   107‚Üí   107‚Üí    osc1.frequency.setValueAtTime(freq, now);
   108‚Üí   108‚Üí   108‚Üí    // Slight pitch envelope (attack transient ‚Äî string stretches then settles)
   109‚Üí   109‚Üí   109‚Üí    osc1.frequency.setValueAtTime(freq * 1.005, now);
   110‚Üí   110‚Üí   110‚Üí    osc1.frequency.exponentialRampToValueAtTime(freq, now + 0.02);
   111‚Üí   111‚Üí   111‚Üí    gain1.gain.setValueAtTime(0, now);
   112‚Üí   112‚Üí   112‚Üí    gain1.gain.linearRampToValueAtTime(0.32, now + 0.004);  // Fast pluck attack
   113‚Üí   113‚Üí   113‚Üí    gain1.gain.exponentialRampToValueAtTime(0.12, now + 0.08);
   114‚Üí   114‚Üí   114‚Üí    gain1.gain.exponentialRampToValueAtTime(0.001, now + duration);
   115‚Üí   115‚Üí   115‚Üí
   116‚Üí   116‚Üí   116‚Üí    // ‚îÄ‚îÄ 2nd harmonic (octave above, softer) ‚îÄ‚îÄ
   117‚Üí   117‚Üí   117‚Üí    const osc2 = ctx.createOscillator();
   118‚Üí   118‚Üí   118‚Üí    const gain2 = ctx.createGain();
   119‚Üí   119‚Üí   119‚Üí    osc2.type = 'sine';
   120‚Üí   120‚Üí   120‚Üí    osc2.frequency.setValueAtTime(freq * 2, now);
   121‚Üí   121‚Üí   121‚Üí    gain2.gain.setValueAtTime(0, now);
   122‚Üí   122‚Üí   122‚Üí    gain2.gain.linearRampToValueAtTime(0.06, now + 0.003);
   123‚Üí   123‚Üí   123‚Üí    gain2.gain.exponentialRampToValueAtTime(0.001, now + duration * 0.4);
   124‚Üí   124‚Üí   124‚Üí
   125‚Üí   125‚Üí   125‚Üí    // ‚îÄ‚îÄ 3rd harmonic ‚îÄ‚îÄ
   126‚Üí   126‚Üí   126‚Üí    const osc3 = ctx.createOscillator();
   127‚Üí   127‚Üí   127‚Üí    const gain3 = ctx.createGain();
   128‚Üí   128‚Üí   128‚Üí    osc3.type = 'sine';
   129‚Üí   129‚Üí   129‚Üí    osc3.frequency.setValueAtTime(freq * 3, now);
   130‚Üí   130‚Üí   130‚Üí    gain3.gain.setValueAtTime(0, now);
   131‚Üí   131‚Üí   131‚Üí    gain3.gain.linearRampToValueAtTime(0.025, now + 0.003);
   132‚Üí   132‚Üí   132‚Üí    gain3.gain.exponentialRampToValueAtTime(0.001, now + duration * 0.25);
   133‚Üí   133‚Üí   133‚Üí
   134‚Üí   134‚Üí   134‚Üí    // ‚îÄ‚îÄ Body resonance (low-pass filtered sub-harmonic) ‚îÄ‚îÄ
   135‚Üí   135‚Üí   135‚Üí    const oscBody = ctx.createOscillator();
   136‚Üí   136‚Üí   136‚Üí    const gainBody = ctx.createGain();
   137‚Üí   137‚Üí   137‚Üí    const filterBody = ctx.createBiquadFilter();
   138‚Üí   138‚Üí   138‚Üí    oscBody.type = 'sine';
   139‚Üí   139‚Üí   139‚Üí    oscBody.frequency.setValueAtTime(freq * 0.5, now);
   140‚Üí   140‚Üí   140‚Üí    filterBody.type = 'lowpass';
   141‚Üí   141‚Üí   141‚Üí    filterBody.frequency.setValueAtTime(300, now);
   142‚Üí   142‚Üí   142‚Üí    filterBody.Q.setValueAtTime(1, now);
   143‚Üí   143‚Üí   143‚Üí    gainBody.gain.setValueAtTime(0, now);
   144‚Üí   144‚Üí   144‚Üí    gainBody.gain.linearRampToValueAtTime(0.04, now + 0.01);
   145‚Üí   145‚Üí   145‚Üí    gainBody.gain.exponentialRampToValueAtTime(0.001, now + duration * 0.4);
   146‚Üí   146‚Üí   146‚Üí
   147‚Üí   147‚Üí   147‚Üí    // ‚îÄ‚îÄ Brightness (sawtooth ‚Üí lowpass, simulates pick attack brightness) ‚îÄ‚îÄ
   148‚Üí   148‚Üí   148‚Üí    const oscBright = ctx.createOscillator();
   149‚Üí   149‚Üí   149‚Üí    const gainBright = ctx.createGain();
   150‚Üí   150‚Üí   150‚Üí    const filterBright = ctx.createBiquadFilter();
   151‚Üí   151‚Üí   151‚Üí    oscBright.type = 'sawtooth';
   152‚Üí   152‚Üí   152‚Üí    oscBright.frequency.setValueAtTime(freq, now);
   153‚Üí   153‚Üí   153‚Üí    filterBright.type = 'lowpass';
   154‚Üí   154‚Üí   154‚Üí    filterBright.frequency.setValueAtTime(4000, now);           // Bright attack
   155‚Üí   155‚Üí   155‚Üí    filterBright.frequency.exponentialRampToValueAtTime(800, now + 0.15); // Dulls quickly
   156‚Üí   156‚Üí   156‚Üí    filterBright.Q.setValueAtTime(0.7, now);
   157‚Üí   157‚Üí   157‚Üí    gainBright.gain.setValueAtTime(0, now);
   158‚Üí   158‚Üí   158‚Üí    gainBright.gain.linearRampToValueAtTime(0.05, now + 0.002);
   159‚Üí   159‚Üí   159‚Üí    gainBright.gain.exponentialRampToValueAtTime(0.001, now + duration * 0.3);
   160‚Üí   160‚Üí   160‚Üí
   161‚Üí   161‚Üí   161‚Üí    // ‚îÄ‚îÄ Pluck noise burst (very short, simulates pick/finger noise) ‚îÄ‚îÄ
   162‚Üí   162‚Üí   162‚Üí    const noiseLen = 0.015;
   163‚Üí   163‚Üí   163‚Üí    const noiseBuffer = ctx.createBuffer(1, ctx.sampleRate * noiseLen, ctx.sampleRate);
   164‚Üí   164‚Üí   164‚Üí    const noiseData = noiseBuffer.getChannelData(0);
   165‚Üí   165‚Üí   165‚Üí    for (let i = 0; i < noiseData.length; i++) noiseData[i] = (Math.random() * 2 - 1) * 0.3;
   166‚Üí   166‚Üí   166‚Üí    const noiseSource = ctx.createBufferSource();
   167‚Üí   167‚Üí   167‚Üí    noiseSource.buffer = noiseBuffer;
   168‚Üí   168‚Üí   168‚Üí    const noiseGain = ctx.createGain();
   169‚Üí   169‚Üí   169‚Üí    const noiseFilter = ctx.createBiquadFilter();
   170‚Üí   170‚Üí   170‚Üí    noiseFilter.type = 'bandpass';
   171‚Üí   171‚Üí   171‚Üí    noiseFilter.frequency.setValueAtTime(2000 + stringIdx * 500, now); // Higher strings = brighter noise
   172‚Üí   172‚Üí   172‚Üí    noiseFilter.Q.setValueAtTime(2, now);
   173‚Üí   173‚Üí   173‚Üí    noiseGain.gain.setValueAtTime(0.12, now);
   174‚Üí   174‚Üí   174‚Üí    noiseGain.gain.exponentialRampToValueAtTime(0.001, now + noiseLen);
   175‚Üí   175‚Üí   175‚Üí
   176‚Üí   176‚Üí   176‚Üí    // Connect everything
   177‚Üí   177‚Üí   177‚Üí    osc1.connect(gain1); gain1.connect(master);
   178‚Üí   178‚Üí   178‚Üí    osc2.connect(gain2); gain2.connect(master);
   179‚Üí   179‚Üí   179‚Üí    osc3.connect(gain3); gain3.connect(master);
   180‚Üí   180‚Üí   180‚Üí    oscBody.connect(filterBody); filterBody.connect(gainBody); gainBody.connect(master);
   181‚Üí   181‚Üí   181‚Üí    oscBright.connect(filterBright); filterBright.connect(gainBright); gainBright.connect(master);
   182‚Üí   182‚Üí   182‚Üí    noiseSource.connect(noiseFilter); noiseFilter.connect(noiseGain); noiseGain.connect(master);
   183‚Üí   183‚Üí   183‚Üí
   184‚Üí   184‚Üí   184‚Üí    // Start all
   185‚Üí   185‚Üí   185‚Üí    const stopTime = now + duration + 0.1;
   186‚Üí   186‚Üí   186‚Üí    osc1.start(now); osc2.start(now); osc3.start(now); oscBody.start(now); oscBright.start(now); noiseSource.start(now);
   187‚Üí   187‚Üí   187‚Üí    osc1.stop(stopTime); osc2.stop(stopTime); osc3.stop(stopTime); oscBody.stop(stopTime); oscBright.stop(stopTime);
   188‚Üí   188‚Üí   188‚Üí  } catch (e) {}
   189‚Üí   189‚Üí   189‚Üí}

   192‚Üí   192‚Üí   192‚Üífunction playMetronomeClick(accent: boolean) {

   197‚Üí   197‚Üí   197‚Üí

   199‚Üí   199‚Üí   199‚Üí    const osc = ctx.createOscillator();
   200‚Üí   200‚Üí   200‚Üí    const gain = ctx.createGain();

   202‚Üí   202‚Üí   202‚Üí    osc.frequency.setValueAtTime(accent ? 1567.98 : 1046.50, now); // G6 (accent) / C6 (normal)
   203‚Üí   203‚Üí   203‚Üí    gain.gain.setValueAtTime(accent ? 0.35 : 0.18, now);
   204‚Üí   204‚Üí   204‚Üí    gain.gain.exponentialRampToValueAtTime(0.001, now + 0.04);
   205‚Üí   205‚Üí   205‚Üí    osc.connect(gain);
   206‚Üí   206‚Üí   206‚Üí
   207‚Üí   207‚Üí   207‚Üí    // Click noise burst (the "clack" of the wood block)
   208‚Üí   208‚Üí   208‚Üí    const noiseLen = 0.008;
   209‚Üí   209‚Üí   209‚Üí    const noiseBuffer = ctx.createBuffer(1, ctx.sampleRate * noiseLen, ctx.sampleRate);
   210‚Üí   210‚Üí   210‚Üí    const noiseData = noiseBuffer.getChannelData(0);
   211‚Üí   211‚Üí   211‚Üí    for (let i = 0; i < noiseData.length; i++) noiseData[i] = (Math.random() * 2 - 1);
   212‚Üí   212‚Üí   212‚Üí    const noiseSource = ctx.createBufferSource();
   213‚Üí   213‚Üí   213‚Üí    noiseSource.buffer = noiseBuffer;
   214‚Üí   214‚Üí   214‚Üí    const noiseGain = ctx.createGain();
   215‚Üí   215‚Üí   215‚Üí    const noiseFilter = ctx.createBiquadFilter();
   216‚Üí   216‚Üí   216‚Üí    noiseFilter.type = 'highpass';
   217‚Üí   217‚Üí   217‚Üí    noiseFilter.frequency.setValueAtTime(accent ? 3000 : 2000, now);
   218‚Üí   218‚Üí   218‚Üí    noiseGain.gain.setValueAtTime(accent ? 0.15 : 0.07, now);
   219‚Üí   219‚Üí   219‚Üí    noiseGain.gain.exponentialRampToValueAtTime(0.001, now + noiseLen);
   220‚Üí   220‚Üí   220‚Üí    noiseSource.connect(noiseFilter); noiseFilter.connect(noiseGain);
   221‚Üí   221‚Üí   221‚Üí
   222‚Üí   222‚Üí   222‚Üí    // Resonance body
   223‚Üí   223‚Üí   223‚Üí    const oscBody = ctx.createOscillator();
   224‚Üí   224‚Üí   224‚Üí    const gainBody = ctx.createGain();
   225‚Üí   225‚Üí   225‚Üí    oscBody.type = 'sine';
   226‚Üí   226‚Üí   226‚Üí    oscBody.frequency.setValueAtTime(accent ? 800 : 600, now);
   227‚Üí   227‚Üí   227‚Üí    gainBody.gain.setValueAtTime(accent ? 0.08 : 0.04, now);
   228‚Üí   228‚Üí   228‚Üí    gainBody.gain.exponentialRampToValueAtTime(0.001, now + 0.03);
   229‚Üí   229‚Üí   229‚Üí    oscBody.connect(gainBody);
   230‚Üí   230‚Üí   230‚Üí
