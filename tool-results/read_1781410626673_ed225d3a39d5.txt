   656→// ─── VARIATION SYSTEM ───
   657→
   658→export interface ExerciseVariation {
   659→  id: string;
   660→  typeId: ExerciseType;
   661→  variantName: string;
   662→  suffix: string;
   663→  modifier: 'standard' | 'extended' | 'inverted' | 'reverse' | 'compact';
   664→}
   665→
   666→export const EXERCISE_VARIATIONS: ExerciseVariation[] = buildVariations();
   667→
   668→function buildVariations(): ExerciseVariation[] {
   669→  const variations: ExerciseVariation[] = [];
   670→  const seenIds = new Set<string>();
   671→  const allTypes = Object.keys(EXERCISE_TYPES) as ExerciseType[];
   672→
   673→  function addVariation(v: ExerciseVariation) {
   674→    if (!seenIds.has(v.id)) {
   675→      seenIds.add(v.id);
   676→      variations.push(v);
   677→    }
   678→  }
   679→
   680→  for (const typeId of allTypes) {
   681→    // Every type gets standard
   682→    addVariation({
   683→      id: `${typeId}-standard`,
   684→      typeId,
   685→      variantName: 'Standard',
   686→      suffix: 'Std',
   687→      modifier: 'standard',
   688→    });
   689→
   690→    // Scale runs: extended + reverse
   691→    if (typeId.startsWith('scale-') || typeId === 'reverse-pentatonic') {
   692→      addVariation({ id: `${typeId}-extended`, typeId, variantName: 'Extended', suffix: 'Ext', modifier: 'extended' });
   693→      addVariation({ id: `${typeId}-reverse`, typeId, variantName: 'Reverse', suffix: 'Rev', modifier: 'reverse' });
   694→    }
   695→
   696→    // Sequences: extended + compact
   697→    if (['thirds', 'fourths', 'fifths', 'sixths', 'sevenths', 'sequence-3', 'sequence-4', 'sequence-5', 'sequence-6', 'sequence-7', 'pedal-tone', 'pedal-tone-inv', 'scale-fragment-3', 'scale-fragment-4', 'scale-skip-note'].includes(typeId)) {
   698→      addVariation({ id: `${typeId}-extended`, typeId, variantName: 'Extended', suffix: 'Ext', modifier: 'extended' });
   699→      addVariation({ id: `${typeId}-compact`, typeId, variantName: 'Compact', suffix: 'Cmp', modifier: 'compact' });
   700→    }
   701→
   702→    // Shapes: inverted + compact
   703→    if (['triads', 'triads-inv1', 'triads-inv2', 'arpeggios', 'octave-shapes', 'double-stops', 'shell-voicings', 'dyads', 'chord-scale', 'power-shapes'].includes(typeId)) {
   704→      addVariation({ id: `${typeId}-inverted`, typeId, variantName: 'Inverted', suffix: 'Inv', modifier: 'inverted' });
   705→      addVariation({ id: `${typeId}-compact`, typeId, variantName: 'Compact', suffix: 'Cmp', modifier: 'compact' });
   706→    }
   707→
   708→    // Technique: extended + reverse
   709→    if (['string-skip', 'lateral-run', 'diagonal', 'position-shift', 'pentatonic-run', 'economy-picking', 'spider-walk', 'interval-jump', 'hammer-pull', 'slide-exercise', 'cross-string', 'speed-burst'].includes(typeId)) {
   710→      addVariation({ id: `${typeId}-extended`, typeId, variantName: 'Extended', suffix: 'Ext', modifier: 'extended' });
   711→      addVariation({ id: `${typeId}-reverse`, typeId, variantName: 'Reverse', suffix: 'Rev', modifier: 'reverse' });
   712→    }
   713→
   714→    // Connections: extended
   715→    if (['connecting', 'position-sweep', 'caged-cycle', 'caged-run'].includes(typeId)) {
   716→      addVariation({ id: `${typeId}-extended`, typeId, variantName: 'Extended', suffix: 'Ext', modifier: 'extended' });
   717→    }
   718→
   719→    // Intervals: inverted + compact
   720→    if (typeId.startsWith('intervals-')) {
   721→      addVariation({ id: `${typeId}-inverted`, typeId, variantName: 'Inverted', suffix: 'Inv', modifier: 'inverted' });
   722→      addVariation({ id: `${typeId}-compact`, typeId, variantName: 'Compact', suffix: 'Cmp', modifier: 'compact' });
   723→    }
   724→
   725→    // Arpeggios: extended + inverted
   726→    if (['arp-sweep', 'arp-string-skip', 'arp-triad-up', 'arp-seventh', 'arp-inverted', 'arp-connect'].includes(typeId)) {
   727→      addVariation({ id: `${typeId}-extended`, typeId, variantName: 'Extended', suffix: 'Ext', modifier: 'extended' });
   728→      addVariation({ id: `${typeId}-inverted`, typeId, variantName: 'Inverted', suffix: 'Inv', modifier: 'inverted' });
   729→    }
   730→
   731→    // Warmups: extended + compact
   732→    if (['chromatic-warmup', 'spider-warmup', 'finger-gym', 'string-cross-warmup'].includes(typeId)) {
   733→      addVariation({ id: `${typeId}-extended`, typeId, variantName: 'Extended', suffix: 'Ext', modifier: 'extended' });
   734→      addVariation({ id: `${typeId}-compact`, typeId, variantName: 'Compact', suffix: 'Cmp', modifier: 'compact' });
   735→    }
   736→
   737→    // Rhythm: extended + compact
   738→    if (typeId.startsWith('rhythm-')) {
   739→      addVariation({ id: `${typeId}-extended`, typeId, variantName: 'Extended', suffix: 'Ext', modifier: 'extended' });
   740→      addVariation({ id: `${typeId}-compact`, typeId, variantName: 'Compact', suffix: 'Cmp', modifier: 'compact' });
   741→    }
   742→
   743→    // Bending: extended
   744→    if (typeId.startsWith('bend-') || typeId === 'vibrato-control') {

  1033→function dedupNotes(notes: FretboardNote[]): FretboardNote[] {
  1034→  const seen = new Set<string>();
  1035→  return notes.filter(n => {
  1036→    const k = `${n.string}-${n.fret}`;
  1037→    if (seen.has(k)) return false;
  1038→    seen.add(k);
  1039→    return true;
  1040→  });
  1041→}

  1043→export function sortNotesAscending(notes: FretboardNote[]): FretboardNote[] {
  1044→  return [...notes].sort((a, b) => {
  1045→    const pitchA = (5 - a.string) * 5 + a.fret;
  1046→    const pitchB = (5 - b.string) * 5 + b.fret;
  1047→    if (pitchA !== pitchB) return pitchA - pitchB;
  1048→    return a.fret - b.fret;
  1049→  });
  1050→}
  1051→
  1052→function sortNotesDescending(notes: FretboardNote[]): FretboardNote[] {
  1053→  return [...notes].sort((a, b) => {
  1054→    const pitchA = (5 - a.string) * 5 + a.fret;
  1055→    const pitchB = (5 - b.string) * 5 + b.fret;
  1056→    if (pitchA !== pitchB) return pitchB - pitchA;
  1057→    return b.fret - a.fret;
  1058→  });
  1059→}

  1065→function generateTabs(notes: ExerciseNote[], fretCount: number = 20): string[] {
  1066→  const stringNames = ['e', 'B', 'G', 'D', 'A', 'E'];
  1067→  const tabs: string[] = stringNames.map(s => s + '|');
  1068→
  1069→  const maxDisplayNotes = Math.min(notes.length, 20);
  1070→  const displayNotes = notes.slice(0, maxDisplayNotes);
  1071→
  1072→  for (const note of displayNotes) {
  1073→    const stringRow = 5 - note.string;
  1074→    for (let row = 0; row < 6; row++) {
  1075→      if (row === stringRow) {
  1076→        const fretStr = note.fret.toString();
  1077→        tabs[row] += '-' + fretStr + '-';
  1078→      } else {
  1079→        tabs[row] += '---';
  1080→      }
  1081→    }
  1082→  }
  1083→
  1084→  const maxLen = Math.max(...tabs.map(t => t.length));
  1085→  for (let i = 0; i < tabs.length; i++) {
  1086→    while (tabs[i].length < maxLen) tabs[i] += '-';
  1087→  }
  1088→
  1089→  return tabs;
  1090→}

  1092→function makeExercise(type: ExerciseType, notes: FretboardNote[], fallbackNotes: FretboardNote[]): Exercise {
  1093→  const usedNotes = notes.length >= 3 ? notes : fallbackNotes;

  1097→    description: EXERCISE_TYPES[type].description,
  1098→    type,
  1099→    notes: exerciseNotes,

  1101→  };
  1102→}

  1219→    return makeExercise('pedal-tone', sorted, sorted);

  1221→  const pedalNotes: FretboardNote[] = [];

  1224→    pedalNotes.push(primaryRoot);
  1225→    pedalNotes.push(note);
  1226→  }
  1227→  pedalNotes.push(primaryRoot);
  1228→  return makeExercise('pedal-tone', pedalNotes, sorted);
  1229→}

  1251→function generateTriads(key: NoteName, scaleId: string, startFret: number, endFret: number): Exercise {

  1256→    const groupNotes = sorted.filter(n => n.string >= stringGroup && n.string <= stringGroup + 2);
  1257→    const roots = groupNotes.filter(n => n.isRoot);
  1258→    for (const root of roots) {
  1259→      triadNotes.push(root);
  1260→      const third = groupNotes.find(n => n.string >= root.string && n.string <= root.string + 2 && (n.interval === 3 || n.interval === 4) && n !== root);
  1261→      if (third) triadNotes.push(third);
  1262→      const fifth = groupNotes.find(n => n.string >= root.string && n.string <= root.string + 2 && n.interval === 7 && n !== root);
  1263→      if (fifth) triadNotes.push(fifth);
  1264→    }
  1265→  }
  1266→  return makeExercise('triads', triadNotes, sorted);
  1267→}

  1305→function generateArpeggios(key: NoteName, scaleId: string, startFret: number, endFret: number): Exercise {
  1306→  const notes = getScaleOnFretboard(key, scaleId, startFret, endFret);
  1307→  const sorted = sortNotesAscending(dedupNotes(notes));
  1308→  const arpNotes: FretboardNote[] = [];
  1309→  const seen = new Set<string>();
  1310→
  1311→  for (let i = 0; i < sorted.length; i++) {
  1312→    const root = sorted[i];
  1313→    const key = `${root.string}-${root.fret}`;
  1314→    if (seen.has(key)) continue;
  1315→
  1316→    const thirdCandidates = sorted.filter(n =>
  1317→      !seen.has(`${n.string}-${n.fret}`) &&
  1318→      n !== root &&
  1319→      (n.interval === (root.interval + 3) % 12 || n.interval === (root.interval + 4) % 12) &&
  1320→      Math.abs(n.string - root.string) <= 2
  1321→    );
  1322→    const fifthCandidates = sorted.filter(n =>
  1323→      !seen.has(`${n.string}-${n.fret}`) &&
  1324→      n !== root &&
  1325→      (n.interval === (root.interval + 7) % 12) &&
  1326→      Math.abs(n.string - root.string) <= 3
  1327→    );
  1328→
  1329→    if (root.isRoot || i % 2 === 0) {
  1330→      seen.add(key);
  1331→      arpNotes.push(root);
  1332→      if (thirdCandidates.length > 0) {
  1333→        const third = thirdCandidates[0];
  1334→        seen.add(`${third.string}-${third.fret}`);
  1335→        arpNotes.push(third);
  1336→      }
  1337→      if (fifthCandidates.length > 0) {
  1338→        const fifth = fifthCandidates[0];
  1339→        seen.add(`${fifth.string}-${fifth.fret}`);
  1340→        arpNotes.push(fifth);
  1341→      }
  1342→    }
  1343→  }
  1344→

  1376→function generateDoubleStops(key: NoteName, scaleId: string, startFret: number, endFret: number): Exercise {

  1392→      }
  1393→    }
  1394→  }

  1400→function generateShellVoicings(key: NoteName, scaleId: string, startFret: number, endFret: number): Exercise {
  1401→  const notes = getScaleOnFretboard(key, scaleId, startFret, endFret);
  1402→  const sorted = sortNotesAscending(dedupNotes(notes));
  1403→  const shellNotes: FretboardNote[] = [];
  1404→  const seen = new Set<string>();
  1405→
  1406→  // Shell voicings: Root + 3rd + 7th (or b7th)
  1407→  for (let i = 0; i < sorted.length; i++) {
  1408→    const root = sorted[i];
  1409→    if (!root.isRoot && i % 3 !== 0) continue;
  1410→    const rootKey = `${root.string}-${root.fret}`;
  1411→    if (seen.has(rootKey)) continue;
  1412→
  1413→    // Find 3rd and 7th on nearby strings
  1414→    const third = sorted.find(n =>
  1415→      !seen.has(`${n.string}-${n.fret}`) &&
  1416→      n !== root &&
  1417→      (n.interval === 3 || n.interval === 4) &&
  1418→      Math.abs(n.string - root.string) <= 3 &&
  1419→      n.string > root.string
  1420→    );
  1421→    const seventh = sorted.find(n =>
  1422→      !seen.has(`${n.string}-${n.fret}`) &&
  1423→      n !== root &&
  1424→      n !== third &&
  1425→      (n.interval === 10 || n.interval === 11) &&
  1426→      Math.abs(n.string - root.string) <= 4 &&
  1427→      n.string > root.string
  1428→    );
  1429→
  1430→    if (third || seventh) {
  1431→      seen.add(rootKey);
  1432→      shellNotes.push(root);
  1433→      if (third) {
  1434→        seen.add(`${third.string}-${third.fret}`);
  1435→        shellNotes.push(third);
  1436→      }
  1437→      if (seventh) {
  1438→        seen.add(`${seventh.string}-${seventh.fret}`);
  1439→        shellNotes.push(seventh);
  1440→      }
  1441→    }
  1442→  }
  1443→
  1444→  return makeExercise('shell-voicings', shellNotes, sorted);
