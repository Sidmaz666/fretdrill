   509→function fretboardToExerciseNote(fn: FretboardNote, seqNum?: number): ExerciseNote {
   510→  return {
   511→    string: fn.string,

   516→    sequenceNumber: seqNum,
   517→  };
   518→}

   520→function dedupNotes(notes: FretboardNote[]): FretboardNote[] {
   521→  const seen = new Set<string>();
   522→  return notes.filter(n => {
   523→    const k = `${n.string}-${n.fret}`;
   524→    if (seen.has(k)) return false;
   525→    seen.add(k);
   526→    return true;
   527→  });
   528→}

   530→export function sortNotesAscending(notes: FretboardNote[]): FretboardNote[] {
   531→  return [...notes].sort((a, b) => {
   532→    const pitchA = (5 - a.string) * 5 + a.fret;
   533→    const pitchB = (5 - b.string) * 5 + b.fret;
   534→    if (pitchA !== pitchB) return pitchA - pitchB;
   535→    return a.fret - b.fret;
   536→  });
   537→}
   538→
   539→function sortNotesDescending(notes: FretboardNote[]): FretboardNote[] {
   540→  return [...notes].sort((a, b) => {
   541→    const pitchA = (5 - a.string) * 5 + a.fret;
   542→    const pitchB = (5 - b.string) * 5 + b.fret;
   543→    if (pitchA !== pitchB) return pitchB - pitchA;
   544→    return b.fret - a.fret;
   545→  });
   546→}

   548→function addSequenceNumbers(notes: ExerciseNote[]): ExerciseNote[] {
   549→  return notes.map((n, i) => ({ ...n, sequenceNumber: i + 1 }));
   550→}

   552→function generateTabs(notes: ExerciseNote[], fretCount: number = 20): string[] {
   553→  const stringNames = ['e', 'B', 'G', 'D', 'A', 'E'];
   554→  const tabs: string[] = stringNames.map(s => s + '|');
   555→
   556→  const maxDisplayNotes = Math.min(notes.length, 20);
   557→  const displayNotes = notes.slice(0, maxDisplayNotes);
   558→
   559→  for (const note of displayNotes) {
   560→    const stringRow = 5 - note.string;
   561→    for (let row = 0; row < 6; row++) {
   562→      if (row === stringRow) {
   563→        const fretStr = note.fret.toString();
   564→        tabs[row] += '-' + fretStr + '-';
   565→      } else {
   566→        tabs[row] += '---';
   567→      }
   568→    }
   569→  }
   570→
   571→  const maxLen = Math.max(...tabs.map(t => t.length));
   572→  for (let i = 0; i < tabs.length; i++) {
   573→    while (tabs[i].length < maxLen) tabs[i] += '-';
   574→  }
   575→
   576→  return tabs;
   577→}

   579→function makeExercise(type: ExerciseType, notes: FretboardNote[], fallbackNotes: FretboardNote[]): Exercise {
   580→  const usedNotes = notes.length >= 3 ? notes : fallbackNotes;
   581→  const exerciseNotes = addSequenceNumbers(usedNotes.map(fretboardToExerciseNote));
   582→  return {
   583→    name: EXERCISE_TYPES[type].name,
   584→    description: EXERCISE_TYPES[type].description,
   585→    type,
   586→    notes: exerciseNotes,
   587→    tabs: generateTabs(exerciseNotes),
   588→  };
   589→}

   748→      if (third) triadNotes.push(third);

   750→      if (fifth) triadNotes.push(fifth);
   751→    }
   752→  }
   753→  return makeExercise('triads', triadNotes, sorted);

   792→function generateArpeggios(key: NoteName, scaleId: string, startFret: number, endFret: number): Exercise {
   793→  const notes = getScaleOnFretboard(key, scaleId, startFret, endFret);
   794→  const sorted = sortNotesAscending(dedupNotes(notes));
   795→  const arpNotes: FretboardNote[] = [];
   796→  const seen = new Set<string>();
   797→
   798→  for (let i = 0; i < sorted.length; i++) {
   799→    const root = sorted[i];
   800→    const key = `${root.string}-${root.fret}`;
   801→    if (seen.has(key)) continue;
   802→
   803→    const thirdCandidates = sorted.filter(n =>
   804→      !seen.has(`${n.string}-${n.fret}`) &&
   805→      n !== root &&
   806→      (n.interval === (root.interval + 3) % 12 || n.interval === (root.interval + 4) % 12) &&
   807→      Math.abs(n.string - root.string) <= 2
   808→    );
   809→    const fifthCandidates = sorted.filter(n =>
   810→      !seen.has(`${n.string}-${n.fret}`) &&
   811→      n !== root &&
   812→      (n.interval === (root.interval + 7) % 12) &&
   813→      Math.abs(n.string - root.string) <= 3
   814→    );
   815→
   816→    if (root.isRoot || i % 2 === 0) {
   817→      seen.add(key);
   818→      arpNotes.push(root);
   819→      if (thirdCandidates.length > 0) {
   820→        const third = thirdCandidates[0];
   821→        seen.add(`${third.string}-${third.fret}`);
   822→        arpNotes.push(third);
   823→      }
   824→      if (fifthCandidates.length > 0) {
   825→        const fifth = fifthCandidates[0];
   826→        seen.add(`${fifth.string}-${fifth.fret}`);
   827→        arpNotes.push(fifth);
   828→      }
   829→    }
   830→  }
   831→
   832→  return makeExercise('arpeggios', arpNotes, sorted);
   833→}

   887→function generateShellVoicings(key: NoteName, scaleId: string, startFret: number, endFret: number): Exercise {
   888→  const notes = getScaleOnFretboard(key, scaleId, startFret, endFret);
   889→  const sorted = sortNotesAscending(dedupNotes(notes));
   890→  const shellNotes: FretboardNote[] = [];
   891→  const seen = new Set<string>();
   892→
   893→  // Shell voicings: Root + 3rd + 7th (or b7th)
   894→  for (let i = 0; i < sorted.length; i++) {
   895→    const root = sorted[i];
   896→    if (!root.isRoot && i % 3 !== 0) continue;
   897→    const rootKey = `${root.string}-${root.fret}`;
   898→    if (seen.has(rootKey)) continue;
   899→
   900→    // Find 3rd and 7th on nearby strings
   901→    const third = sorted.find(n =>
   902→      !seen.has(`${n.string}-${n.fret}`) &&
   903→      n !== root &&
   904→      (n.interval === 3 || n.interval === 4) &&
   905→      Math.abs(n.string - root.string) <= 3 &&
   906→      n.string > root.string
   907→    );
   908→    const seventh = sorted.find(n =>
   909→      !seen.has(`${n.string}-${n.fret}`) &&
   910→      n !== root &&
   911→      n !== third &&
   912→      (n.interval === 10 || n.interval === 11) &&
   913→      Math.abs(n.string - root.string) <= 4 &&
   914→      n.string > root.string
   915→    );
   916→
   917→    if (third || seventh) {
   918→      seen.add(rootKey);
   919→      shellNotes.push(root);
   920→      if (third) {
   921→        seen.add(`${third.string}-${third.fret}`);
   922→        shellNotes.push(third);
   923→      }
   924→      if (seventh) {
   925→        seen.add(`${seventh.string}-${seventh.fret}`);
   926→        shellNotes.push(seventh);
   927→      }
   928→    }
   929→  }
   930→
   931→  return makeExercise('shell-voicings', shellNotes, sorted);
   932→}

   934→function generateDyads(key: NoteName, scaleId: string, startFret: number, endFret: number): Exercise {
   935→  const notes = getScaleOnFretboard(key, scaleId, startFret, endFret);
   936→  const sorted = sortNotesAscending(dedupNotes(notes));
   937→  const dyadNotes: FretboardNote[] = [];
   938→  const seen = new Set<string>();
   939→
   940→  // Two-note combinations on adjacent strings
   941→  for (let s = 0; s < 5; s++) {
   942→    const lowerNotes = sorted.filter(n => n.string === s);
   943→    const higherNotes = sorted.filter(n => n.string === s + 1);
   944→    for (const ln of lowerNotes) {
   945→      // Find closest note on next string
   946→      const closest = higherNotes.reduce((best: FretboardNote | null, hn) => {
   947→        if (seen.has(`${hn.string}-${hn.fret}`)) return best;
   948→        const dist = Math.abs(hn.fret - ln.fret);
   949→        if (!best || dist < Math.abs(best.fret - ln.fret)) return hn;
   950→        return best;
   951→      }, null);
   952→      if (closest) {
   953→        const lk = `${ln.string}-${ln.fret}`;
   954→        const ck = `${closest.string}-${closest.fret}`;
   955→        if (!seen.has(lk)) { seen.add(lk); dyadNotes.push(ln); }
   956→        if (!seen.has(ck)) { seen.add(ck); dyadNotes.push(closest); }
   957→      }
   958→    }
   959→  }
   960→
   961→  return makeExercise('dyads', dyadNotes, sorted);
   962→}

   964→function generateChordScale(key: NoteName, scaleId: string, startFret: number, endFret: number): Exercise {
   965→  const notes = getScaleOnFretboard(key, scaleId, startFret, endFret);
   966→  const sorted = sortNotesAscending(dedupNotes(notes));
   967→  const chordNotes: FretboardNote[] = [];
   968→  const seen = new Set<string>();
   969→
   970→  // Build triads from each scale degree
   971→  const scale = SCALES[scaleId];
   972→  if (!scale) return makeExercise('chord-scale', sorted, sorted);
   973→
   974→  const intervals = scale.intervals;
   975→  for (let deg = 0; deg < intervals.length; deg++) {
   976→    const rootInterval = intervals[deg];
   977→    const thirdInterval = intervals[(deg + 2) % intervals.length];
   978→    const fifthInterval = intervals[(deg + 4) % intervals.length];
   979→
   980→    const rootCandidates = sorted.filter(n => n.interval === rootInterval && !seen.has(`${n.string}-${n.fret}`));
   981→    const thirdCandidates = sorted.filter(n => n.interval === thirdInterval && !seen.has(`${n.string}-${n.fret}`));
   982→    const fifthCandidates = sorted.filter(n => n.interval === fifthInterval && !seen.has(`${n.string}-${n.fret}`));
   983→
   984→    if (rootCandidates.length > 0) {
   985→      const root = rootCandidates[0];
   986→      seen.add(`${root.string}-${root.fret}`);
   987→      chordNotes.push(root);
   988→
   989→      const third = thirdCandidates.find(n => Math.abs(n.string - root.string) <= 2);
   990→      if (third) {
   991→        seen.add(`${third.string}-${third.fret}`);
   992→        chordNotes.push(third);
   993→      }
   994→      const fifth = fifthCandidates.find(n => Math.abs(n.string - root.string) <= 3);
   995→      if (fifth) {
   996→        seen.add(`${fifth.string}-${fifth.fret}`);
   997→        chordNotes.push(fifth);
   998→      }
   999→    }
  1000→  }
  1001→
  1002→  return makeExercise('chord-scale', chordNotes, sorted);
  1003→}

  1469→function generateArpTriadUp(key: NoteName, scaleId: string, startFret: number, endFret: number): Exercise {

  1471→  const sorted = sortNotesAscending(dedupNotes(notes));
  1472→  const arpNotes: FretboardNote[] = [];
  1473→
  1474→  // Triad arpeggio ascending through each scale degree
  1475→  const scale = SCALES[scaleId];
  1476→  if (!scale) return makeExercise('arp-triad-up', sorted, sorted);
  1477→
  1478→  for (let deg = 0; deg < scale.intervals.length; deg++) {
  1479→    const rootInterval = scale.intervals[deg];
  1480→    const thirdInterval = scale.intervals[(deg + 2) % scale.intervals.length];
  1481→    const fifthInterval = scale.intervals[(deg + 4) % scale.intervals.length];
  1482→
  1483→    const root = sorted.find(n => n.interval === rootInterval && !arpNotes.includes(n));
  1484→    const third = sorted.find(n => n.interval === thirdInterval && n !== root && !arpNotes.includes(n));
  1485→    const fifth = sorted.find(n => n.interval === fifthInterval && n !== root && n !== third && !arpNotes.includes(n));
  1486→
  1487→    if (root) arpNotes.push(root);
  1488→    if (third) arpNotes.push(third);
  1489→    if (fifth) arpNotes.push(fifth);
