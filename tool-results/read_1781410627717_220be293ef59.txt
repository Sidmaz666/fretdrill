    74→// ─── AUDIO ENGINE ───
    75→// Proper standard guitar tuning frequencies (Hz) — E2, A2, D3, G3, B3, E4
    76→const STRING_FREQUENCIES = [82.41, 110.00, 146.83, 196.00, 246.94, 329.63];
    77→
    78→let audioCtx: AudioContext | null = null;
    79→let masterCompressor: DynamicsCompressorNode | null = null;
    80→function getAudioCtx() {
    81→  if (!audioCtx) audioCtx = new AudioContext();
    82→  return audioCtx;
    83→}
    84→
    85→// Singleton master compressor
    86→function getMaster(ctx: AudioContext): DynamicsCompressorNode {
    87→  if (!masterCompressor) {
    88→    const comp = ctx.createDynamicsCompressor();
    89→    comp.threshold.setValueAtTime(-20, ctx.currentTime);
    90→    comp.knee.setValueAtTime(10, ctx.currentTime);
    91→    comp.ratio.setValueAtTime(4, ctx.currentTime);
    92→    comp.attack.setValueAtTime(0.003, ctx.currentTime);
    93→    comp.release.setValueAtTime(0.15, ctx.currentTime);
    94→    comp.connect(ctx.destination);
    95→    masterCompressor = comp;
    96→  }
    97→  return masterCompressor;
    98→}

   100→function playGuitarNote(stringIdx: number, fret: number, duration: number = 0.5) {
   101→  try {
   102→    const ctx = getAudioCtx();
   103→    if (ctx.state === 'suspended') ctx.resume();
   104→    const master = getMaster(ctx);
   105→    const baseFreq = STRING_FREQUENCIES[stringIdx];
   106→    const freq = baseFreq * Math.pow(2, fret / 12);
   107→    const now = ctx.currentTime;
   108→
   109→    // ── Fundamental (pulse wave — close to plucked string) ──
   110→    const osc1 = ctx.createOscillator();
   111→    const gain1 = ctx.createGain();
   112→    osc1.type = 'triangle';
   113→    osc1.frequency.setValueAtTime(freq, now);
   114→    // Slight pitch envelope (attack transient — string stretches then settles)
   115→    osc1.frequency.setValueAtTime(freq * 1.005, now);
   116→    osc1.frequency.exponentialRampToValueAtTime(freq, now + 0.02);
   117→    gain1.gain.setValueAtTime(0, now);
   118→    gain1.gain.linearRampToValueAtTime(0.32, now + 0.004);  // Fast pluck attack
   119→    gain1.gain.exponentialRampToValueAtTime(0.12, now + 0.08);
   120→    gain1.gain.exponentialRampToValueAtTime(0.001, now + duration);
   121→
   122→    // ── 2nd harmonic (octave above, softer) ──
   123→    const osc2 = ctx.createOscillator();
   124→    const gain2 = ctx.createGain();
   125→    osc2.type = 'sine';
   126→    osc2.frequency.setValueAtTime(freq * 2, now);
   127→    gain2.gain.setValueAtTime(0, now);
   128→    gain2.gain.linearRampToValueAtTime(0.06, now + 0.003);
   129→    gain2.gain.exponentialRampToValueAtTime(0.001, now + duration * 0.4);
   130→
   131→    // ── 3rd harmonic ──
   132→    const osc3 = ctx.createOscillator();
   133→    const gain3 = ctx.createGain();
   134→    osc3.type = 'sine';
   135→    osc3.frequency.setValueAtTime(freq * 3, now);
   136→    gain3.gain.setValueAtTime(0, now);
   137→    gain3.gain.linearRampToValueAtTime(0.025, now + 0.003);
   138→    gain3.gain.exponentialRampToValueAtTime(0.001, now + duration * 0.25);
   139→
   140→    // ── Body resonance (low-pass filtered sub-harmonic) ──
   141→    const oscBody = ctx.createOscillator();
   142→    const gainBody = ctx.createGain();
   143→    const filterBody = ctx.createBiquadFilter();
   144→    oscBody.type = 'sine';
   145→    oscBody.frequency.setValueAtTime(freq * 0.5, now);
   146→    filterBody.type = 'lowpass';
   147→    filterBody.frequency.setValueAtTime(300, now);
   148→    filterBody.Q.setValueAtTime(1, now);
   149→    gainBody.gain.setValueAtTime(0, now);
   150→    gainBody.gain.linearRampToValueAtTime(0.04, now + 0.01);
   151→    gainBody.gain.exponentialRampToValueAtTime(0.001, now + duration * 0.4);
   152→
   153→    // ── Brightness (sawtooth → lowpass, simulates pick attack brightness) ──
   154→    const oscBright = ctx.createOscillator();
   155→    const gainBright = ctx.createGain();
   156→    const filterBright = ctx.createBiquadFilter();
   157→    oscBright.type = 'sawtooth';
   158→    oscBright.frequency.setValueAtTime(freq, now);
   159→    filterBright.type = 'lowpass';
   160→    filterBright.frequency.setValueAtTime(4000, now);           // Bright attack
   161→    filterBright.frequency.exponentialRampToValueAtTime(800, now + 0.15); // Dulls quickly
   162→    filterBright.Q.setValueAtTime(0.7, now);
   163→    gainBright.gain.setValueAtTime(0, now);
   164→    gainBright.gain.linearRampToValueAtTime(0.05, now + 0.002);
   165→    gainBright.gain.exponentialRampToValueAtTime(0.001, now + duration * 0.3);
   166→
   167→    // ── Pluck noise burst (very short, simulates pick/finger noise) ──
   168→    const noiseLen = 0.015;
   169→    const noiseBuffer = ctx.createBuffer(1, ctx.sampleRate * noiseLen, ctx.sampleRate);
   170→    const noiseData = noiseBuffer.getChannelData(0);
   171→    for (let i = 0; i < noiseData.length; i++) noiseData[i] = (Math.random() * 2 - 1) * 0.3;
   172→    const noiseSource = ctx.createBufferSource();
   173→    noiseSource.buffer = noiseBuffer;
   174→    const noiseGain = ctx.createGain();
   175→    const noiseFilter = ctx.createBiquadFilter();
   176→    noiseFilter.type = 'bandpass';
   177→    noiseFilter.frequency.setValueAtTime(2000 + stringIdx * 500, now); // Higher strings = brighter noise
   178→    noiseFilter.Q.setValueAtTime(2, now);
   179→    noiseGain.gain.setValueAtTime(0.12, now);
   180→    noiseGain.gain.exponentialRampToValueAtTime(0.001, now + noiseLen);
   181→
   182→    // Connect everything
   183→    osc1.connect(gain1); gain1.connect(master);
   184→    osc2.connect(gain2); gain2.connect(master);
   185→    osc3.connect(gain3); gain3.connect(master);
   186→    oscBody.connect(filterBody); filterBody.connect(gainBody); gainBody.connect(master);
   187→    oscBright.connect(filterBright); filterBright.connect(gainBright); gainBright.connect(master);
   188→    noiseSource.connect(noiseFilter); noiseFilter.connect(noiseGain); noiseGain.connect(master);
   189→
   190→    // Start all
   191→    const stopTime = now + duration + 0.1;
   192→    osc1.start(now); osc2.start(now); osc3.start(now); oscBody.start(now); oscBright.start(now); noiseSource.start(now);
   193→    osc1.stop(stopTime); osc2.stop(stopTime); osc3.stop(stopTime); oscBody.stop(stopTime); oscBright.stop(stopTime);
   194→  } catch (e) {}
   195→}

   197→// Professional metronome — wood block sound
   198→function playMetronomeClick(accent: boolean) {
   199→  try {
   200→    const ctx = getAudioCtx();
   201→    if (ctx.state === 'suspended') ctx.resume();
   202→    const now = ctx.currentTime;
   203→
   204→    // Tone component — short sine burst like a wood block
   205→    const osc = ctx.createOscillator();
   206→    const gain = ctx.createGain();
   207→    osc.type = 'sine';
   208→    osc.frequency.setValueAtTime(accent ? 1567.98 : 1046.50, now); // G6 (accent) / C6 (normal)
   209→    gain.gain.setValueAtTime(accent ? 0.35 : 0.18, now);
   210→    gain.gain.exponentialRampToValueAtTime(0.001, now + 0.04);
   211→    osc.connect(gain);
   212→
   213→    // Click noise burst (the "clack" of the wood block)
   214→    const noiseLen = 0.008;
   215→    const noiseBuffer = ctx.createBuffer(1, ctx.sampleRate * noiseLen, ctx.sampleRate);
   216→    const noiseData = noiseBuffer.getChannelData(0);
   217→    for (let i = 0; i < noiseData.length; i++) noiseData[i] = (Math.random() * 2 - 1);
   218→    const noiseSource = ctx.createBufferSource();
   219→    noiseSource.buffer = noiseBuffer;
   220→    const noiseGain = ctx.createGain();
   221→    const noiseFilter = ctx.createBiquadFilter();
   222→    noiseFilter.type = 'highpass';
   223→    noiseFilter.frequency.setValueAtTime(accent ? 3000 : 2000, now);
   224→    noiseGain.gain.setValueAtTime(accent ? 0.15 : 0.07, now);
   225→    noiseGain.gain.exponentialRampToValueAtTime(0.001, now + noiseLen);
   226→    noiseSource.connect(noiseFilter); noiseFilter.connect(noiseGain);
   227→
   228→    // Resonance body
   229→    const oscBody = ctx.createOscillator();
   230→    const gainBody = ctx.createGain();
   231→    oscBody.type = 'sine';
   232→    oscBody.frequency.setValueAtTime(accent ? 800 : 600, now);
   233→    gainBody.gain.setValueAtTime(accent ? 0.08 : 0.04, now);
   234→    gainBody.gain.exponentialRampToValueAtTime(0.001, now + 0.03);
   235→    oscBody.connect(gainBody);
   236→
   237→    const master = getMaster(ctx);
   238→    gain.connect(master); noiseGain.connect(master); gainBody.connect(master);
   239→
   240→    osc.start(now); noiseSource.start(now); oscBody.start(now);
   241→    osc.stop(now + 0.05); oscBody.stop(now + 0.04);
   242→  } catch (e) {}
   243→}

   376→// ─── INTERVAL COLOR ───
   377→function getIntervalColor(label: string): string {
   378→  if (label === 'R') return '#9b3939';
   379→  if (label.includes('\u266D3') || label.includes('\u266D2')) return '#6b4a7a';
   380→  if (label === '3' || label === '2') return '#4a7a4a';
   381→  if (label === '4') return '#4a5a8a';
   382→  if (label === '5') return '#4a7a7a';
   383→  if (label.includes('5')) return '#4a7a7a';
   384→  if (label.includes('6')) return '#8a6a3a';
   385→  if (label.includes('7')) return '#7a4a6a';
   386→  if (label.includes('\u266F') || label.includes('\u266D')) return '#6b4a7a';
   387→  return '#5a5a6a';
   388→}
