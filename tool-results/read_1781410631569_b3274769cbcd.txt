    75→    75→// Proper standard guitar tuning frequencies (Hz) — E2, A2, D3, G3, B3, E4
    76→    76→const STRING_FREQUENCIES = [82.41, 110.00, 146.83, 196.00, 246.94, 329.63];
    77→    77→
    78→    78→let audioCtx: AudioContext | null = null;
    79→    79→let masterCompressor: DynamicsCompressorNode | null = null;
    80→    80→function getAudioCtx() {
    81→    81→  if (!audioCtx) audioCtx = new AudioContext();
    82→    82→  return audioCtx;
    83→    83→}

    85→    85→// Singleton master compressor
    86→    86→function getMaster(ctx: AudioContext): DynamicsCompressorNode {
    87→    87→  if (!masterCompressor) {
    88→    88→    const comp = ctx.createDynamicsCompressor();
    89→    89→    comp.threshold.setValueAtTime(-20, ctx.currentTime);
    90→    90→    comp.knee.setValueAtTime(10, ctx.currentTime);
    91→    91→    comp.ratio.setValueAtTime(4, ctx.currentTime);
    92→    92→    comp.attack.setValueAtTime(0.003, ctx.currentTime);
    93→    93→    comp.release.setValueAtTime(0.15, ctx.currentTime);
    94→    94→    comp.connect(ctx.destination);
    95→    95→    masterCompressor = comp;
    96→    96→  }
    97→    97→  return masterCompressor;
    98→    98→}

   100→   100→function playGuitarNote(stringIdx: number, fret: number, duration: number = 0.5) {
   101→   101→  try {
   102→   102→    const ctx = getAudioCtx();
   103→   103→    if (ctx.state === 'suspended') ctx.resume();
   104→   104→    const master = getMaster(ctx);
   105→   105→    const baseFreq = STRING_FREQUENCIES[stringIdx];
   106→   106→    const freq = baseFreq * Math.pow(2, fret / 12);
   107→   107→    const now = ctx.currentTime;
   108→   108→
   109→   109→    // ── Fundamental (pulse wave — close to plucked string) ──
   110→   110→    const osc1 = ctx.createOscillator();
   111→   111→    const gain1 = ctx.createGain();
   112→   112→    osc1.type = 'triangle';
   113→   113→    osc1.frequency.setValueAtTime(freq, now);
   114→   114→    // Slight pitch envelope (attack transient — string stretches then settles)
   115→   115→    osc1.frequency.setValueAtTime(freq * 1.005, now);
   116→   116→    osc1.frequency.exponentialRampToValueAtTime(freq, now + 0.02);
   117→   117→    gain1.gain.setValueAtTime(0, now);
   118→   118→    gain1.gain.linearRampToValueAtTime(0.32, now + 0.004);  // Fast pluck attack
   119→   119→    gain1.gain.exponentialRampToValueAtTime(0.12, now + 0.08);
   120→   120→    gain1.gain.exponentialRampToValueAtTime(0.001, now + duration);
   121→   121→
   122→   122→    // ── 2nd harmonic (octave above, softer) ──
   123→   123→    const osc2 = ctx.createOscillator();
   124→   124→    const gain2 = ctx.createGain();
   125→   125→    osc2.type = 'sine';
   126→   126→    osc2.frequency.setValueAtTime(freq * 2, now);
   127→   127→    gain2.gain.setValueAtTime(0, now);
   128→   128→    gain2.gain.linearRampToValueAtTime(0.06, now + 0.003);
   129→   129→    gain2.gain.exponentialRampToValueAtTime(0.001, now + duration * 0.4);
   130→   130→
   131→   131→    // ── 3rd harmonic ──
   132→   132→    const osc3 = ctx.createOscillator();
   133→   133→    const gain3 = ctx.createGain();
   134→   134→    osc3.type = 'sine';
   135→   135→    osc3.frequency.setValueAtTime(freq * 3, now);
   136→   136→    gain3.gain.setValueAtTime(0, now);
   137→   137→    gain3.gain.linearRampToValueAtTime(0.025, now + 0.003);
   138→   138→    gain3.gain.exponentialRampToValueAtTime(0.001, now + duration * 0.25);
   139→   139→
   140→   140→    // ── Body resonance (low-pass filtered sub-harmonic) ──
   141→   141→    const oscBody = ctx.createOscillator();
   142→   142→    const gainBody = ctx.createGain();
   143→   143→    const filterBody = ctx.createBiquadFilter();
   144→   144→    oscBody.type = 'sine';
   145→   145→    oscBody.frequency.setValueAtTime(freq * 0.5, now);
   146→   146→    filterBody.type = 'lowpass';
   147→   147→    filterBody.frequency.setValueAtTime(300, now);
   148→   148→    filterBody.Q.setValueAtTime(1, now);
   149→   149→    gainBody.gain.setValueAtTime(0, now);
   150→   150→    gainBody.gain.linearRampToValueAtTime(0.04, now + 0.01);
   151→   151→    gainBody.gain.exponentialRampToValueAtTime(0.001, now + duration * 0.4);
   152→   152→
   153→   153→    // ── Brightness (sawtooth → lowpass, simulates pick attack brightness) ──
   154→   154→    const oscBright = ctx.createOscillator();
   155→   155→    const gainBright = ctx.createGain();
   156→   156→    const filterBright = ctx.createBiquadFilter();
   157→   157→    oscBright.type = 'sawtooth';
   158→   158→    oscBright.frequency.setValueAtTime(freq, now);
   159→   159→    filterBright.type = 'lowpass';
   160→   160→    filterBright.frequency.setValueAtTime(4000, now);           // Bright attack
   161→   161→    filterBright.frequency.exponentialRampToValueAtTime(800, now + 0.15); // Dulls quickly
   162→   162→    filterBright.Q.setValueAtTime(0.7, now);
   163→   163→    gainBright.gain.setValueAtTime(0, now);
   164→   164→    gainBright.gain.linearRampToValueAtTime(0.05, now + 0.002);
   165→   165→    gainBright.gain.exponentialRampToValueAtTime(0.001, now + duration * 0.3);
   166→   166→
   167→   167→    // ── Pluck noise burst (very short, simulates pick/finger noise) ──
   168→   168→    const noiseLen = 0.015;
   169→   169→    const noiseBuffer = ctx.createBuffer(1, ctx.sampleRate * noiseLen, ctx.sampleRate);
   170→   170→    const noiseData = noiseBuffer.getChannelData(0);
   171→   171→    for (let i = 0; i < noiseData.length; i++) noiseData[i] = (Math.random() * 2 - 1) * 0.3;
   172→   172→    const noiseSource = ctx.createBufferSource();
   173→   173→    noiseSource.buffer = noiseBuffer;
   174→   174→    const noiseGain = ctx.createGain();
   175→   175→    const noiseFilter = ctx.createBiquadFilter();
   176→   176→    noiseFilter.type = 'bandpass';
   177→   177→    noiseFilter.frequency.setValueAtTime(2000 + stringIdx * 500, now); // Higher strings = brighter noise
   178→   178→    noiseFilter.Q.setValueAtTime(2, now);
   179→   179→    noiseGain.gain.setValueAtTime(0.12, now);
   180→   180→    noiseGain.gain.exponentialRampToValueAtTime(0.001, now + noiseLen);
   181→   181→
   182→   182→    // Connect everything
   183→   183→    osc1.connect(gain1); gain1.connect(master);
   184→   184→    osc2.connect(gain2); gain2.connect(master);
   185→   185→    osc3.connect(gain3); gain3.connect(master);
   186→   186→    oscBody.connect(filterBody); filterBody.connect(gainBody); gainBody.connect(master);
   187→   187→    oscBright.connect(filterBright); filterBright.connect(gainBright); gainBright.connect(master);
   188→   188→    noiseSource.connect(noiseFilter); noiseFilter.connect(noiseGain); noiseGain.connect(master);
   189→   189→
   190→   190→    // Start all
   191→   191→    const stopTime = now + duration + 0.1;
   192→   192→    osc1.start(now); osc2.start(now); osc3.start(now); oscBody.start(now); oscBright.start(now); noiseSource.start(now);
   193→   193→    osc1.stop(stopTime); osc2.stop(stopTime); osc3.stop(stopTime); oscBody.stop(stopTime); oscBright.stop(stopTime);
   194→   194→  } catch (e) {}
   195→   195→}

   197→   197→// Professional metronome — wood block sound
   198→   198→function playMetronomeClick(accent: boolean) {
   199→   199→  try {
   200→   200→    const ctx = getAudioCtx();
   201→   201→    if (ctx.state === 'suspended') ctx.resume();
   202→   202→    const now = ctx.currentTime;
   203→   203→
   204→   204→    // Tone component — short sine burst like a wood block
   205→   205→    const osc = ctx.createOscillator();
   206→   206→    const gain = ctx.createGain();
   207→   207→    osc.type = 'sine';
   208→   208→    osc.frequency.setValueAtTime(accent ? 1567.98 : 1046.50, now); // G6 (accent) / C6 (normal)
   209→   209→    gain.gain.setValueAtTime(accent ? 0.35 : 0.18, now);
   210→   210→    gain.gain.exponentialRampToValueAtTime(0.001, now + 0.04);
   211→   211→    osc.connect(gain);
   212→   212→
   213→   213→    // Click noise burst (the "clack" of the wood block)
   214→   214→    const noiseLen = 0.008;
   215→   215→    const noiseBuffer = ctx.createBuffer(1, ctx.sampleRate * noiseLen, ctx.sampleRate);
   216→   216→    const noiseData = noiseBuffer.getChannelData(0);
   217→   217→    for (let i = 0; i < noiseData.length; i++) noiseData[i] = (Math.random() * 2 - 1);
   218→   218→    const noiseSource = ctx.createBufferSource();
   219→   219→    noiseSource.buffer = noiseBuffer;
   220→   220→    const noiseGain = ctx.createGain();
   221→   221→    const noiseFilter = ctx.createBiquadFilter();
   222→   222→    noiseFilter.type = 'highpass';
   223→   223→    noiseFilter.frequency.setValueAtTime(accent ? 3000 : 2000, now);
   224→   224→    noiseGain.gain.setValueAtTime(accent ? 0.15 : 0.07, now);
   225→   225→    noiseGain.gain.exponentialRampToValueAtTime(0.001, now + noiseLen);
   226→   226→    noiseSource.connect(noiseFilter); noiseFilter.connect(noiseGain);
   227→   227→
   228→   228→    // Resonance body
   229→   229→    const oscBody = ctx.createOscillator();
   230→   230→    const gainBody = ctx.createGain();
   231→   231→    oscBody.type = 'sine';
   232→   232→    oscBody.frequency.setValueAtTime(accent ? 800 : 600, now);
   233→   233→    gainBody.gain.setValueAtTime(accent ? 0.08 : 0.04, now);
   234→   234→    gainBody.gain.exponentialRampToValueAtTime(0.001, now + 0.03);
   235→   235→    oscBody.connect(gainBody);
   236→   236→
   237→   237→    const master = getMaster(ctx);
   238→   238→    gain.connect(master); noiseGain.connect(master); gainBody.connect(master);
   239→   239→
   240→   240→    osc.start(now); noiseSource.start(now); oscBody.start(now);
   241→   241→    osc.stop(now + 0.05); oscBody.stop(now + 0.04);
   242→   242→  } catch (e) {}
   243→   243→}

   361→   361→        gain.gain.setValueAtTime(vol * 0.3, now); gain.gain.exponentialRampToValueAtTime(0.001, now + 0.05);
   362→   362→        const lp = ctx.createBiquadFilter(); lp.type = 'lowpass'; lp.frequency.setValueAtTime(3000, now);
   363→   363→        osc.connect(lp); lp.connect(gain); gain.connect(master);
   364→   364→        osc.start(now); osc.stop(now + 0.06);
   365→   365→        const res = ctx.createOscillator(); const rGain = ctx.createGain();
   366→   366→        res.type = 'sine'; res.frequency.setValueAtTime(accent ? 400 : 300, now);
   367→   367→        rGain.gain.setValueAtTime(vol * 0.06, now); rGain.gain.exponentialRampToValueAtTime(0.001, now + 0.08);
   368→   368→        res.connect(rGain); rGain.connect(master);
   369→   369→        res.start(now); res.stop(now + 0.1);
   370→   370→        break;
   371→   371→      }
   372→   372→    }
   373→   373→  } catch (e) {}
   374→   374→}
