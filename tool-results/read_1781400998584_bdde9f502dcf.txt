    95тЖТ    95тЖТexport function sortNotesAscending(notes: FretboardNote[]): FretboardNote[] {
    96тЖТ    96тЖТ  return [...notes].sort((a, b) => {
    97тЖТ    97тЖТ    // Calculate pitch: each string is roughly 5 semitones apart
    98тЖТ    98тЖТ    const pitchA = (5 - a.string) * 5 + a.fret; // high strings = higher pitch
    99тЖТ    99тЖТ    const pitchB = (5 - b.string) * 5 + b.fret;
   100тЖТ   100тЖТ    if (pitchA !== pitchB) return pitchA - pitchB;
   101тЖТ   101тЖТ    return a.fret - b.fret;
   102тЖТ   102тЖТ  });
   103тЖТ   103тЖТ}

   105тЖТ   105тЖТfunction sortNotesDescending(notes: FretboardNote[]): FretboardNote[] {
   106тЖТ   106тЖТ  return [...notes].sort((a, b) => {
   107тЖТ   107тЖТ    const pitchA = (5 - a.string) * 5 + a.fret;
   108тЖТ   108тЖТ    const pitchB = (5 - b.string) * 5 + b.fret;
   109тЖТ   109тЖТ    if (pitchA !== pitchB) return pitchB - pitchA;
   110тЖТ   110тЖТ    return b.fret - a.fret;
   111тЖТ   111тЖТ  });
   112тЖТ   112тЖТ}

   120тЖТ   120тЖТfunction generateTabs(notes: ExerciseNote[], fretCount: number = 20): string[] {
   121тЖТ   121тЖТ  const stringNames = ['e', 'B', 'G', 'D', 'A', 'E'];
   122тЖТ   122тЖТ  const tabs: string[] = stringNames.map(s => s + '|');
   123тЖТ   123тЖТ  
   124тЖТ   124тЖТ  const maxDisplayNotes = Math.min(notes.length, 20);
   125тЖТ   125тЖТ  const displayNotes = notes.slice(0, maxDisplayNotes);
   126тЖТ   126тЖТ
   127тЖТ   127тЖТ  for (const note of displayNotes) {
   128тЖТ   128тЖТ    const stringRow = 5 - note.string;
   129тЖТ   129тЖТ    for (let row = 0; row < 6; row++) {
   130тЖТ   130тЖТ      if (row === stringRow) {
   131тЖТ   131тЖТ        const fretStr = note.fret.toString();
   132тЖТ   132тЖТ        tabs[row] += '-' + fretStr + '-';
   133тЖТ   133тЖТ      } else {
   134тЖТ   134тЖТ        tabs[row] += '---';
   135тЖТ   135тЖТ      }
   136тЖТ   136тЖТ    }
   137тЖТ   137тЖТ  }
   138тЖТ   138тЖТ
   139тЖТ   139тЖТ  const maxLen = Math.max(...tabs.map(t => t.length));
   140тЖТ   140тЖТ  for (let i = 0; i < tabs.length; i++) {
   141тЖТ   141тЖТ    while (tabs[i].length < maxLen) tabs[i] += '-';
   142тЖТ   142тЖТ  }
   143тЖТ   143тЖТ
   144тЖТ   144тЖТ  return tabs;
   145тЖТ   145тЖТ}

   240тЖТ   240тЖТ  }

   242тЖТ   242тЖТ  for (const note of sorted) {

   255тЖТ   255тЖТ  const triadNotes: FretboardNote[] = [];
   256тЖТ   256тЖТ  for (let stringGroup = 0; stringGroup < 4; stringGroup++) {
   257тЖТ   257тЖТ    const groupNotes = sorted.filter(n => n.string >= stringGroup && n.string <= stringGroup + 2);
   258тЖТ   258тЖТ    const roots = groupNotes.filter(n => n.isRoot);
   259тЖТ   259тЖТ    for (const root of roots) {
   260тЖТ   260тЖТ      triadNotes.push(root);
   261тЖТ   261тЖТ      const third = groupNotes.find(n => n.string >= root.string && n.string <= root.string + 2 && (n.interval === 3 || n.interval === 4) && n !== root);
   262тЖТ   262тЖТ      if (third) triadNotes.push(third);
   263тЖТ   263тЖТ      const fifth = groupNotes.find(n => n.string >= root.string && n.string <= root.string + 2 && n.interval === 7 && n !== root);
   264тЖТ   264тЖТ      if (fifth) triadNotes.push(fifth);
   265тЖТ   265тЖТ    }
   266тЖТ   266тЖТ  }
   267тЖТ   267тЖТ  const exerciseNotes = addSequenceNumbers(triadNotes.map(fretboardToExerciseNote));
   268тЖТ   268тЖТ  return { name: EXERCISE_TYPES['triads'].name, description: EXERCISE_TYPES['triads'].description, type: 'triads', notes: exerciseNotes, tabs: generateTabs(exerciseNotes) };

   271тЖТ   271тЖТfunction generateArpeggios(key: NoteName, scaleId: string, startFret: number, endFret: number): Exercise {
   272тЖТ   272тЖТ  const notes = getScaleOnFretboard(key, scaleId, startFret, endFret);
   273тЖТ   273тЖТ  const sorted = sortNotesAscending(notes);
   274тЖТ   274тЖТ  const arpNotes: FretboardNote[] = [];
   275тЖТ   275тЖТ  const seen = new Set<string>();
   276тЖТ   276тЖТ
   277тЖТ   277тЖТ  // Build arpeggios from each scale degree: root, 3rd/2nd, 5th/4th
   278тЖТ   278тЖТ  const scale = { intervals: notes.length > 0 ? [...new Set(notes.map(n => n.interval))].sort((a, b) => a - b) : [] };
   279тЖТ   279тЖТ  
   280тЖТ   280тЖТ  // For each unique interval/degree, find a triad-like grouping
   281тЖТ   281тЖТ  for (let i = 0; i < sorted.length; i++) {
   282тЖТ   282тЖТ    const root = sorted[i];
   283тЖТ   283тЖТ    const key = `${root.string}-${root.fret}`;
   284тЖТ   284тЖТ    if (seen.has(key)) continue;
   285тЖТ   285тЖТ
   286тЖТ   286тЖТ    // Find 3rd and 5th relative to this note within the scale
   287тЖТ   287тЖТ    const thirdCandidates = sorted.filter(n => 
   288тЖТ   288тЖТ      !seen.has(`${n.string}-${n.fret}`) &&
   289тЖТ   289тЖТ      n !== root &&

   557тЖТ   557тЖТ  const notes = getScaleOnFretboard(key, scaleId, startFret, endFret);

   559тЖТ   559тЖТ  const jumpNotes: FretboardNote[] = [];
   560тЖТ   560тЖТ  
   561тЖТ   561тЖТ  if (sorted.length < 4) {
   562тЖТ   562тЖТ    return generateAscendingScale(key, scaleId, startFret, endFret);
   563тЖТ   563тЖТ  }
   564тЖТ   564тЖТ
   565тЖТ   565тЖТ  // Jump pattern: play low note, high note, medium low, medium high
   566тЖТ   566тЖТ  // Creates interval jumps for ear training
   567тЖТ   567тЖТ  const mid = Math.floor(sorted.length / 2);
   568тЖТ   568тЖТ  const patterns = [
   569тЖТ   569тЖТ    0, sorted.length - 1,  // low -> high (large jump)
   570тЖТ   570тЖТ    1, sorted.length - 2,  // slightly less
   571тЖТ   571тЖТ    mid, 0,                // mid -> low
   572тЖТ   572тЖТ    mid + 1, sorted.length - 1, // mid+1 -> high
   573тЖТ   573тЖТ    mid - 1, 1,            // mid-1 -> low+1
   574тЖТ   574тЖТ    sorted.length - 1, mid, // high -> mid
   575тЖТ   575тЖТ    0, mid + 1,            // low -> mid+1
   576тЖТ   576тЖТ  ];
   577тЖТ   577тЖТ  
   578тЖТ   578тЖТ  const seen = new Set<number>();
   579тЖТ   579тЖТ  for (const idx of patterns) {
   580тЖТ   580тЖТ    if (idx >= 0 && idx < sorted.length && !seen.has(idx)) {
   581тЖТ   581тЖТ      seen.add(idx);
   582тЖТ   582тЖТ      jumpNotes.push(sorted[idx]);
   583тЖТ   583тЖТ    }
   584тЖТ   584тЖТ  }
   585тЖТ   585тЖТ  // Add remaining notes
   586тЖТ   586тЖТ  for (let i = 0; i < sorted.length; i++) {
   587тЖТ   587тЖТ    if (!seen.has(i)) {
   588тЖТ   588тЖТ      seen.add(i);
   589тЖТ   589тЖТ      jumpNotes.push(sorted[i]);
   590тЖТ   590тЖТ    }
   591тЖТ   591тЖТ  }
   592тЖТ   592тЖТ
   593тЖТ   593тЖТ  const exerciseNotes = addSequenceNumbers(jumpNotes.map(fretboardToExerciseNote));
   594тЖТ   594тЖТ  return { name: EXERCISE_TYPES['interval-jump'].name, description: EXERCISE_TYPES['interval-jump'].description, type: 'interval-jump', notes: exerciseNotes, tabs: generateTabs(exerciseNotes) };
   595тЖТ   595тЖТ}

   600тЖТ   600тЖТ  type: ExerciseType,

   607тЖТ   607тЖТ  const startFret = position.fretStart;
   608тЖТ   608тЖТ  const endFret = position.fretEnd;

   617тЖТ   617тЖТ    case 'sequence-4': return generateSequence4(key, scaleId, startFret, endFret);
   618тЖТ   618тЖТ    case 'sequence-5': return generateSequence5(key, scaleId, startFret, endFret);
   619тЖТ   619тЖТ    case 'pedal-tone': return generatePedalTone(key, scaleId, startFret, endFret);
   620тЖТ   620тЖТ    case 'triads': return generateTriads(key, scaleId, startFret, endFret);
   621тЖТ   621тЖТ    case 'arpeggios': return generateArpeggios(key, scaleId, startFret, endFret);
   622тЖТ   622тЖТ    case 'octave-shapes': return generateOctaveShapes(key, scaleId, startFret, endFret);
   623тЖТ   623тЖТ    case 'double-stops': return generateDoubleStops(key, scaleId, startFret, endFret);
   624тЖТ   624тЖТ    case 'connecting': return generateConnecting(key, scaleId);
   625тЖТ   625тЖТ    case 'string-skip': return generateStringSkip(key, scaleId, startFret, endFret);
   626тЖТ   626тЖТ    case 'lateral-run': return generateLateralRun(key, scaleId, startFret, endFret);
   627тЖТ   627тЖТ    case 'diagonal': return generateDiagonal(key, scaleId, startFret, endFret);
   628тЖТ   628тЖТ    case 'position-shift': return generatePositionShift(key, scaleId);
   629тЖТ   629тЖТ    case 'pentatonic-run': return generatePentatonicRun(key, scaleId, startFret, endFret);
   630тЖТ   630тЖТ    case 'economy-picking': return generateEconomyPicking(key, scaleId, startFret, endFret);
   631тЖТ   631тЖТ    case 'spider-walk': return generateSpiderWalk(key, scaleId, startFret, endFret);
   632тЖТ   632тЖТ    case 'interval-jump': return generateIntervalJump(key, scaleId, startFret, endFret);
   633тЖТ   633тЖТ    default: return generateAscendingScale(key, scaleId, startFret, endFret);
   634тЖТ   634тЖТ  }
   635тЖТ   635тЖТ}
